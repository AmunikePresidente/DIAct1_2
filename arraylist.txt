    public void listar(){
        DefaultListModel modelo=new DefaultListModel();
        for (int i=0;i<listaAlumnos.size();i++){
            modelo.addElement(listaAlumnos.get(i).getNombre() +", "+ listaAlumnos.get(i).getApellidos()+", "+listaAlumnos.get(i).getEmail()+", "+ listaAlumnos.get(i).getEdad()+", "+listaAlumnos.get(i).getTelefono());
        }
       
        ListaAlumnos.setModel(modelo);     
    }
